// Step 1: Create order
const orderRes = await fetch("/api/orders", {
  method: "POST",
  headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}` },
  body: JSON.stringify({ productId: selectedProductId, locale, gateway: selectedGateway }),
});
if (!orderRes.ok) {
  const err = await orderRes.json();
  if (orderRes.status === 402) setError("Insufficient wallet balance.");
  else setError(err.error ?? "Could not create order.");
  return;
}
const { orderId } = await orderRes.json();

// Step 2: Initiate payment
const initiateRes = await fetch("/api/payment/initiate", {
  method: "POST",
  headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}` },
  body: JSON.stringify({ orderId, gateway: selectedGateway }),
});
if (!initiateRes.ok) {
  setError("Payment could not be started. Please try again.");
  return;
}
const { redirectUrl } = await initiateRes.json();

// Wallet path: no redirect, go directly to order detail
if (!redirectUrl) {
  router.push(`/${locale}/orders/${orderId}`);
} else {
  window.location.href = redirectUrl;
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → checkout form | User-supplied productId and gateway go into order creation — validated server-side |
| Browser → orders pages | Auth token from localStorage sent as Authorization header on every API call |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-10-01 | Spoofing | Checkout/Orders — unauthenticated page access | mitigate | Auth redirect at page mount; API returns 401 without token so data never loads |
| T-03-10-02 | Tampering | Checkout — client sends unknown gateway | mitigate | Server-side Zod enum validation on /api/payment/initiate rejects unknown gateways |
| T-03-10-03 | Information Disclosure | Key reveal shown before payment confirmed | mitigate | KeyRevealBox only rendered when order.status in [PAID, COMPLETED]; PENDING status shows spinner only |
| T-03-10-04 | Tampering | Orders page shows another buyer's orders | mitigate | GET /api/orders filters by auth token userId server-side; client only receives own orders |
</threat_model>
